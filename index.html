function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); // --- ČÁST 1: POSÍLÁNÍ ÚKOLŮ DO MOBILU --- // Pokud aplikace volá ?action=getTasks if (e.parameter.action == "getTasks") { var sheetTasks = ss.getSheetByName("UKOLY"); // Musíš mít list s názvem UKOLY if (!sheetTasks) return ContentService.createTextOutput(JSON.stringify([])).setMimeType(ContentService.MimeType.JSON); var data = sheetTasks.getDataRange().getValues(); var workerName = e.parameter.worker; var activeTasks = []; // Projdeme tabulku úkolů (vynecháme záhlaví na řádku 0) for (var i = 1; i < data.length; i++) { // Sloupec A (0) = Jméno technika, Sloupec D (3) = Status if (data[i][0] == workerName && data[i][3] != "HOTOVO") { activeTasks.push({ place: data[i][1], // Sloupec B: Objekt note: data[i][2], // Sloupec C: Popis úkolu row: i + 1 // Číslo řádku pro budoucí označení jako hotové }); } } return ContentService.createTextOutput(JSON.stringify(activeTasks)) .setMimeType(ContentService.MimeType.JSON); } // --- ČÁST 2: UKLÁDÁNÍ DAT Z MOBILU (Tvůj původní kód) --- // Pokud aplikace jen posílá report o provedené práci var sheetLog = ss.getSheetByName("Sheet1"); // Zkontroluj, zda se tvůj hlavní list jmenuje Sheet1 if (!sheetLog) sheetLog = ss.getSheets()[0]; // Pokud ne, vezme se první list v pořadí sheetLog.appendRow([ e.parameter.time, e.parameter.worker, e.parameter.place, e.parameter.note, e.parameter.gps ]); return ContentService.createTextOutput("OK"); }
