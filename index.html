// Upravená funkce checkTasks s dvěma tlačítky async function checkTasks() { if (!currentUser || !navigator.onLine) return; try { const res = await fetch(`${SCRIPT_URL}?action=getTasks&worker=${encodeURIComponent(currentUser.name)}`); const tasks = await res.json(); const container = document.getElementById('task-container'); if (tasks && tasks.length > 0) { container.innerHTML = tasks.map(t => `
⚠️ NOVÝ ÚKOL
${t.place}
${t.note}
`).join(''); if (tasks.length > lastTaskCount) { playAlarm(); if (navigator.vibrate) navigator.vibrate([1000, 500, 1000]); lastTaskCount = tasks.length; localStorage.setItem('lastTaskCount', lastTaskCount); } } else { container.innerHTML = ""; lastTaskCount = 0; localStorage.setItem('lastTaskCount', 0); } } catch (e) { console.error("Task check failed"); } } // Upravená funkce completeTask pro odeslání stavu async function completeTask(place, status) { const btn = event.target; if (!btn.classList.contains('btn-confirm-stage')) { btn.classList.add('btn-confirm-stage'); const originalText = btn.innerText; btn.innerText = "POTVRĎ!"; setTimeout(() => { if (btn && !btn.disabled) { btn.classList.remove('btn-confirm-stage'); btn.innerText = originalText; } }, 3000); return; } btn.innerText = "SÍŤ..."; btn.disabled = true; try { // Pošleme akci completeTask a přidáme parametr status (Splněno/Nesplněno) await fetch(`${SCRIPT_URL}?action=completeTask&worker=${encodeURIComponent(currentUser.name)}&place=${encodeURIComponent(place)}&status=${encodeURIComponent(status)}`, { mode: 'no-cors' }); const card = btn.closest('.task-card'); if (card) { card.style.opacity = "0"; setTimeout(() => { card.remove(); lastTaskCount = 0; checkTasks(); }, 500); } } catch (e) { alert("Chyba."); btn.disabled = false; btn.innerText = "ZNOVU"; } }
