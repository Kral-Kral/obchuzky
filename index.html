function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var action = e.parameter.action; if (!e || !e.parameter) return ContentService.createTextOutput("Chyba"); // --- 1. NAČÍTÁNÍ ÚKOLŮ --- if (action === "getTasks") { var sheetTasks = ss.getSheetByName("UKOLY"); if (!sheetTasks) return ContentService.createTextOutput("[]").setMimeType(ContentService.MimeType.JSON); var data = sheetTasks.getDataRange().getValues(); var workerName = (e.parameter.worker || "").toString().trim().toLowerCase(); var activeTasks = []; for (var i = 1; i < data.length; i++) { var row = data[i]; var cellWorker = (row[0] || "").toString().trim().toLowerCase(); // Kontrola sloupce D: Pokud tam NENÍ prázdno, úkol už v mobilu NECHCI. var cellStatus = (row[3] || "").toString().trim(); if (cellWorker === workerName && cellStatus === "") { activeTasks.push({ place: row[1] ? row[1].toString() : "Místo neuvedeno", note: row[2] ? row[2].toString() : "" }); } } return ContentService.createTextOutput(JSON.stringify(activeTasks)).setMimeType(ContentService.MimeType.JSON); } // --- 2. ODKLIKNUTÍ ÚKOLU --- if (action === "completeTask") { var sheetTasks = ss.getSheetByName("UKOLY"); var worker = (e.parameter.worker || "").toString().trim().toLowerCase(); var place = (e.parameter.place || "").toString().trim().toLowerCase(); // Tady je ten hlavní fix: Vezmeme jen první slovo (NELZE / HOTOVO) a ignorujeme křížky. var status = (e.parameter.status || "HOTOVO").toString().split(" ")[0].trim().toUpperCase(); if (sheetTasks) { var dataTasks = sheetTasks.getDataRange().getValues(); for (var j = 1; j < dataTasks.length; j++) { var w = (dataTasks[j][0] || "").toString().trim().toLowerCase(); var p = (dataTasks[j][1] || "").toString().trim().toLowerCase(); if (w === worker && p === place) { sheetTasks.getRange(j + 1, 4).setValue(status); // Zapíše čisté NELZE sheetTasks.getRange(j + 1, 5).setValue(new Date()); return ContentService.createTextOutput("OK"); } } } return ContentService.createTextOutput("Nenalezeno"); } // --- 3. ZÁPIS BĚŽNÉ OBCHŮZKY --- var sheetRecords = ss.getSheetByName("Záznamy"); if (!sheetRecords) sheetRecords = ss.getSheets()[0]; sheetRecords.appendRow([new Date(), e.parameter.worker, e.parameter.place, e.parameter.note, e.parameter.gps]); return ContentService.createTextOutput("OK"); }
